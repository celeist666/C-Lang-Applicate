#include<stdio.h>
#include<time.h>

typedef struct date {
	int year = 0, month = 0, day = 0;
} date;


int main(void) {

	time_t curr = time(NULL);
	date da;

	struct tm *d;

	d = localtime(&curr);

	da.year = d->tm_year + 1900;
	da.month = d->tm_mon + 1;
	da.day = d->tm_mday;
	
	printf("현재날짜\n");

	printf("%d년 %d월 %d일\n", da.year, da.month, da.day);


	return 0;
}
