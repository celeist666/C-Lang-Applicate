#include<stdio.h>
#include <string.h>

int RemoveSpace(char* arr) {
	
	int cnt = 0;

	for (; *arr != NULL; arr++)
	//for (; cnt==2; arr++)
	{
		if (*arr == ' ' || *arr == '\t' || *arr == '\v' || *arr == '\n' || *arr == '\f' || *arr == '\r')
		{
			//*arr = *(arr + 1);
			strcpy(arr,arr+1);
			arr--;
			cnt++;
		}
	}
	return cnt;
}



int main(void) {

	int cnt = 0;

	char arr[100];
	printf("문자열을 입력하세요 : ");
	scanf("%[^\n]",arr);

	

	cnt = RemoveSpace(arr);

	printf("결과 문자열 : %s\n\n", arr);
	printf("제거된 공백 문자 개수 : %d\n", cnt);
}
